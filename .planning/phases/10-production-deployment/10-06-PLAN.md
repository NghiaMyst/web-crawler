    ```typescript
    import { BloomFilter } from 'bloom-filters';
    import { connection } from '../connection.js';

    const BLOOM_REDIS_KEY = 'bloom:filter';
    const BLOOM_TTL_SECONDS = 604800; // 7 days

    export let bloomFilter = BloomFilter.create(100000, 0.01);

    export async function loadBloomFilter(): Promise<void> {
      try {
        const serialized = await connection.get(BLOOM_REDIS_KEY);
        if (serialized) {
          bloomFilter = BloomFilter.fromJSON(JSON.parse(serialized));
        }
      } catch {
        // Redis unavailable or corrupt data — start fresh (in-memory filter already set above)
      }
    }

    export async function saveBloomFilter(): Promise<void> {
      const json = JSON.stringify(bloomFilter.saveAsJSON());
      await connection.setex(BLOOM_REDIS_KEY, BLOOM_TTL_SECONDS, json);
    }

    export function isUrlSeen(url: string): boolean {
      return bloomFilter.has(url);
    }

    export function markUrlSeen(url: string): void {
      bloomFilter.add(url);
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| crawler process → Redis | ioredis on internal Docker network — no exposure outside compose network |
| Redis storage → bloom filter state | Bloom filter is a probabilistic structure; serialized JSON contains only bit array + metadata, no URL values |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation |
|-----------|----------|-----------|-------------|------------|
| T-10-21 | T (Tampering) | Attacker writes crafted JSON to bloom:filter Redis key to force false-positives (block legitimate URLs) | accept | Internal network only; Redis not exposed to host (no redis port in docker-compose.prod.yml); bloom filter false-positives only cause skipped crawls, not data loss or security breach |
| T-10-22 | D (Denial of Service) | Redis key missing or corrupt causes loadBloomFilter to throw and crash crawler startup | mitigate | try-catch in loadBloomFilter falls back to fresh in-memory filter; crawler starts even if Redis has no saved state |
| T-10-23 | D (Denial of Service) | saveBloomFilter blocks shutdown if Redis is unreachable | accept | setupGracefulShutdown has a 30-second hard timeout (process.exit(1)); bloom filter loss on unexpected Redis failure is acceptable |
| T-10-24 | I (Information Disclosure) | Bloom filter JSON in Redis leaks which URLs were crawled | accept | bloom:filter key stores a bit array and configuration — no URL strings are stored in the serialized structure (probabilistic design) |
</threat_model>
