    ```html
    </head>
    <body>
      <h1>SignalR Dashboard Feed</h1>
      <p class="meta">
        Dev-only test client (D-05). Phase 9 replaces this with the Next.js client.
        Hub: <code>/hubs/dashboard</code> · Event: <code>NewEntry</code>.
      </p>
      <p>Status: <span id="status" class="status-disconnected">Disconnected</span></p>
      <p class="meta">
        Last received <code>crawledAt</code>: <span id="lastTs">(none)</span>
      </p>
      <h2>Entries</h2>
      <ul id="entries"></ul>

      <!-- RESEARCH D-04 + JS CDN URL: pin to 8.0.0 via cdnjs for reproducibility. -->
      <script src="https://cdnjs.cloudflare.com/ajax/libs/microsoft-signalr/8.0.0/signalr.min.js"></script>
      <script>
        'use strict';

        // lastReceivedAt holds the crawledAt of the most recent NewEntry seen.
        // Used on reconnect to fetch any entries missed during the disconnect gap (D-03).
        let lastReceivedAt = null;

        const statusEl = document.getElementById('status');
        const lastTsEl = document.getElementById('lastTs');
        const listEl = document.getElementById('entries');

        function setStatus(label, cls) {
          statusEl.textContent = label;
          statusEl.className = cls;
        }

        function renderEntry(entry, { backfill = false } = {}) {
          const li = document.createElement('li');
          if (backfill) li.className = 'backfill';
          const ts = entry.crawledAt ? new Date(entry.crawledAt).toISOString() : '(no ts)';
          li.innerHTML =
            '<div class="timestamp">' + (backfill ? '[BACKFILL] ' : '') + ts + ' · ' +
            (entry.category || '(no category)') + ' · key=' + (entry.entryKey || '') + '</div>' +
            '<div>' + JSON.stringify(entry.payload) + '</div>';
          listEl.prepend(li);
        }

        const connection = new signalR.HubConnectionBuilder()
          .withUrl('/hubs/dashboard')
          .withAutomaticReconnect([0, 2000, 10000, 30000])
          .configureLogging(signalR.LogLevel.Information)
          .build();

        // RESEARCH Pitfall 7: register handlers BEFORE calling start().
        connection.on('NewEntry', (entry) => {
          if (entry && entry.crawledAt) {
            lastReceivedAt = entry.crawledAt;
            lastTsEl.textContent = new Date(entry.crawledAt).toISOString();
          }
          renderEntry(entry, { backfill: false });
        });

        // D-03 reconnect gap-fill: pull entries the server emitted while we were disconnected.
        // `from=<lastReceivedAt>` maps to the existing Phase 5 filter e.CrawledAt >= from.Value
        // (apps/api/Endpoints/EntriesEndpoints.cs line 41-42). limit=100 is the endpoint's clamp max.
        connection.onreconnected(async () => {
          setStatus('Connected (recovered)', 'status-connected');
          if (!lastReceivedAt) return;
          try {
            const url = '/api/entries?from=' + encodeURIComponent(lastReceivedAt) + '&limit=100';
            const res = await fetch(url, { credentials: 'include' });
            if (!res.ok) {
              console.error('Gap-fill fetch failed:', res.status, res.statusText);
              return;
            }
            const data = await res.json();
            const missed = Array.isArray(data.items) ? data.items : [];
            // Prepend in chronological order so newest ends up on top after the loop.
            missed.slice().reverse().forEach((e) => renderEntry(e, { backfill: true }));
            console.log('Gap-fill prepended', missed.length, 'entries since', lastReceivedAt);
          } catch (err) {
            console.error('Gap-fill failed:', err);
          }
        });

        connection.onreconnecting(() => setStatus('Reconnecting…', 'status-reconnecting'));
        connection.onclose(() => {
          setStatus('Disconnected', 'status-disconnected');
          // Belt-and-braces: retry manually after withAutomaticReconnect exhausts its attempts.
          setTimeout(start, 5000);
        });

        async function start() {
          try {
            await connection.start();
            setStatus('Connected', 'status-connected');
            console.log('SignalR Connected, connectionId=', connection.connectionId);
          } catch (err) {
            console.error('SignalR start failed:', err);
            setStatus('Disconnected', 'status-disconnected');
            setTimeout(start, 5000);
          }
        }

        start();
      </script>
    </body>
    </html>
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser ↔ /hubs/dashboard | WebSocket with credentials enabled (D-07 CORS); untrusted browser context |
| Browser ↔ /api/entries | HTTP fetch on reconnect; same trust level as existing Phase 5 API surface |
| cdnjs ↔ Browser | Third-party CDN dependency (Cloudflare); subresource integrity not pinned (dev-only page, D-05) |

## STRIDE Threat Register

| Threat | Severity | Mitigation |
|--------|----------|------------|
| CDN compromise injects malicious JS into test page (T-06-10, Tampering) | LOW | Accepted — test page is dev-only (D-05), served only on localhost, never shipped to production. Phase 10 production build will NOT serve this file (documented in this plan; file path is `apps/api/wwwroot/test-signalr.html` which production docker-compose.prod.yml should exclude or this file should be deleted before Phase 10). |
| Static file path traversal reveals secrets (T-06-11, Information Disclosure) | LOW | `UseStaticFiles()` only serves from `wwwroot/` which contains a single file; .NET StaticFileOptions reject `..` segments by default. No secrets live in wwwroot/. |
| Reconnect gap-fill exposes private data to unauthenticated clients (T-06-12, Information Disclosure) | LOW | `/api/entries` is already a public endpoint (Phase 5) with no auth per REQUIREMENTS.md Out-of-Scope. Gap-fill uses the same endpoint with the same visibility. |
| Missing SRI on CDN script (T-06-13, Tampering) | LOW | Accepted per D-05 (dev-only). If this page ever moves toward production, add `integrity="sha384-..."` and `crossorigin="anonymous"` attributes. |
| Manual `NOTIFY` in test harness exposes DB connection string (T-06-14, Information Disclosure) | LOW | Verification step uses `$DATABASE_URL` from developer's local env — same pattern as the rest of the project. Not shipped. |
</threat_model>
