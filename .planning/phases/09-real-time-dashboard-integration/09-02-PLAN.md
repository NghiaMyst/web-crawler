The file must contain:

```typescript
'use client';

import { useEffect, useRef, useState } from 'react';
import { useSignalRContext } from '@/contexts/signalr.context';
import { EntriesTable } from './entries-table';
import type { DataEntry } from '@/types/api';
import { fetchEntriesSince } from '@/lib/api.client';
import { toast } from 'sonner';

const ROW_CAP = 200;

interface LiveEntriesWrapperProps {
  serverEntries: DataEntry[];
  children?: React.ReactNode;
}

export function LiveEntriesWrapper({ serverEntries }: LiveEntriesWrapperProps) {
  const { connection, registerReconnectHandler } = useSignalRContext();
  const [liveEntries, setLiveEntries] = useState<DataEntry[]>([]);
  const lastReceivedAtRef = useRef<string | null>(null);
  const serverEntriesLengthRef = useRef(serverEntries.length);

  // Keep ref in sync with prop
  useEffect(() => {
    serverEntriesLengthRef.current = serverEntries.length;
  }, [serverEntries.length]);

  // Subscribe to NewEntry events
  useEffect(() => {
    if (!connection) return;

    const handler = (entry: DataEntry) => {
      lastReceivedAtRef.current = entry.crawledAt;
      setLiveEntries((prev) => {
        const combined = [entry, ...prev];
        const totalRows = combined.length + serverEntriesLengthRef.current;
        if (totalRows > ROW_CAP) {
          return combined.slice(0, Math.max(0, ROW_CAP - serverEntriesLengthRef.current));
        }
        return combined;
      });
    };

    connection.on('NewEntry', handler);
    return () => {
      connection.off('NewEntry', handler);
    };
  }, [connection]);

  // Register gap recovery handler (fires on reconnect via SignalRProvider)
  useEffect(() => {
    if (!registerReconnectHandler) return;

    const gapRecovery = async () => {
      const since = lastReceivedAtRef.current;
      if (!since) {
        toast.success('Reconnected — no missed entries');
        return;
      }
      try {
        const result = await fetchEntriesSince(since, 50);
        const count = result.items.length;
        if (count > 0) {
          setLiveEntries((prev) => [...result.items, ...prev].slice(0, ROW_CAP));
          toast.success(`Reconnected — loaded ${count} missed entries`);
        } else {
          toast.success('Reconnected — no missed entries');
        }
      } catch {
        // Gap recovery is best-effort; failure is non-fatal
        toast.success('Reconnected — no missed entries');
      }
    };

    registerReconnectHandler(gapRecovery);
    return () => {
      registerReconnectHandler(null);
    };
  }, [registerReconnectHandler]);

  // Merge live entries (newest first) with server entries, enforce cap
  const allEntries = [...liveEntries, ...serverEntries].slice(0, ROW_CAP);

  return <EntriesTable entries={allEntries} />;
}
```

Key design decisions (per CONTEXT.md and RESEARCH.md):
- Connection stored in useState (from 09-01 SignalRProvider), so this effect re-runs when
  connection becomes available
- `connection.on('NewEntry', handler)` with matching `connection.off('NewEntry', handler)`
  using the SAME function reference (per RESEARCH anti-pattern warning)
- `serverEntriesLengthRef` avoids stale closure in the `NewEntry` handler (per RESEARCH Pitfall 4)
- Gap recovery registered via `registerReconnectHandler` (from 09-01) to avoid
  `onreconnected` handler accumulation (per RESEARCH Pitfall 3)
- ROW_CAP = 200 (per D-02)
- Toast messages per UI-SPEC Copywriting Contract:
  - "Reconnected — loaded N missed entries" (N = count from gap recovery)
  - "Reconnected — no missed entries" (when gap recovery returns 0 rows)
