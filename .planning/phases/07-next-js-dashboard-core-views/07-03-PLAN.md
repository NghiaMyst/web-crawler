      ```typescript
      import type { Metadata } from 'next';
      import { fetchEntries, fetchSources } from '@/lib/api.server';
      import { EntriesClient } from '@/components/entries/EntriesClient';
      import { FilterBar } from '@/components/entries/FilterBar';
      import { EntriesEmptyState } from '@/components/entries/EntriesEmptyState';

      export const metadata: Metadata = {
        title: 'Entries — Web Crawler',
      };

      // Force this page to render dynamically (no caching) — it depends on searchParams + live data.
      export const dynamic = 'force-dynamic';

      type SearchParams = Promise<{
        category?: string;
        sourceId?: string;
        from?: string;
        to?: string;
      }>;

      export default async function EntriesPage({
        searchParams,
      }: {
        searchParams: SearchParams;
      }): Promise<React.JSX.Element> {
        const sp = await searchParams;
        const filters = {
          category: sp.category,
          sourceId: sp.sourceId,
          from: sp.from,
          to: sp.to,
          limit: 20,
        };

        const [entriesPage, sources] = await Promise.all([
          fetchEntries(filters),
          fetchSources(),
        ]);

        const sourceMap = Object.fromEntries(sources.map((s) => [s.id, s.displayName]));
        const isFiltered = Boolean(filters.category || filters.sourceId || filters.from || filters.to);

        return (
          <div className="space-y-6">
            <h1 className="text-xl font-semibold text-zinc-900">Entries</h1>
            <FilterBar sources={sources} initialFilters={filters} />
            {entriesPage.items.length === 0 ? (
              <EntriesEmptyState filtered={isFiltered} />
            ) : (
              <EntriesClient
                initialEntries={entriesPage.items}
                initialNextCursor={entriesPage.nextCursor}
                sourceMap={sourceMap}
                filters={filters}
              />
            )}
          </div>
        );
      }
      ```

    Step 1: Create `apps/dashboard/components/entries/FilterBar.tsx` (`'use client'` per D-03 —
    pushes filter values to URL search params, server component re-fetches on URL change):

      ```typescript
      'use client';

      import { useRouter, useSearchParams } from 'next/navigation';
      import { useTransition } from 'react';
      import {
        Select, SelectContent, SelectItem, SelectTrigger, SelectValue,
      } from '@/components/ui/select';
      import { Input } from '@/components/ui/input';
      import { Label } from '@/components/ui/label';
      import type { Source, EntryFilters } from '@/types/api';

      const ALL_VALUE = '__all__';
      const CATEGORIES = ['game', 'football', 'anime', 'manga', 'music'] as const;

      export function FilterBar({
        sources,
        initialFilters,
      }: {
        sources: Source[];
        initialFilters: EntryFilters;
      }): React.JSX.Element {
        const router = useRouter();
        const searchParams = useSearchParams();
        const [isPending, startTransition] = useTransition();

        function updateParam(key: string, value: string | undefined): void {
          const params = new URLSearchParams(searchParams.toString());
          if (value && value !== ALL_VALUE) {
            params.set(key, value);
          } else {
            params.delete(key);
          }
          // Cursor reset on filter change — client accumulator picks up fresh first page.
          params.delete('cursor');
          startTransition(() => {
            router.push(`/entries?${params.toString()}`);
          });
        }

        return (
          <div className="flex flex-col gap-4 md:flex-row md:items-end md:gap-3">
            <div className="flex flex-col gap-1.5">
              <Label htmlFor="filter-category" className="text-xs text-zinc-700">Category</Label>
              <Select
                value={initialFilters.category ?? ALL_VALUE}
                onValueChange={(v) => updateParam('category', v)}
              >
                <SelectTrigger id="filter-category" className="w-full md:w-44">
                  <SelectValue placeholder="All categories" />
                </SelectTrigger>
                <SelectContent>
                  <SelectItem value={ALL_VALUE}>All categories</SelectItem>
                  {CATEGORIES.map((c) => (
                    <SelectItem key={c} value={c}>{c}</SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>

            <div className="flex flex-col gap-1.5">
              <Label htmlFor="filter-source" className="text-xs text-zinc-700">Source</Label>
              <Select
                value={initialFilters.sourceId ?? ALL_VALUE}
                onValueChange={(v) => updateParam('sourceId', v)}
              >
                <SelectTrigger id="filter-source" className="w-full md:w-56">
                  <SelectValue placeholder="All sources" />
                </SelectTrigger>
                <SelectContent>
                  <SelectItem value={ALL_VALUE}>All sources</SelectItem>
                  {sources.map((s) => (
                    <SelectItem key={s.id} value={s.id}>{s.displayName}</SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>

            <div className="flex flex-col gap-1.5">
              <Label htmlFor="filter-from" className="text-xs text-zinc-700">From</Label>
              <Input
                id="filter-from"
                type="date"
                defaultValue={initialFilters.from ? initialFilters.from.slice(0, 10) : ''}
                onChange={(e) => {
                  const v = e.target.value;
                  updateParam('from', v ? new Date(v).toISOString() : undefined);
                }}
                className="w-full md:w-40"
              />
            </div>

            <div className="flex flex-col gap-1.5">
              <Label htmlFor="filter-to" className="text-xs text-zinc-700">To</Label>
              <Input
                id="filter-to"
                type="date"
                defaultValue={initialFilters.to ? initialFilters.to.slice(0, 10) : ''}
                onChange={(e) => {
                  const v = e.target.value;
                  updateParam('to', v ? new Date(v).toISOString() : undefined);
                }}
                className="w-full md:w-40"
              />
            </div>

            {isPending && (
              <span className="text-xs text-zinc-500 self-end pb-2">Updating…</span>
            )}
          </div>
        );
      }
      ```

    Step 2: Create `apps/dashboard/components/entries/EntriesClient.tsx` (`'use client'` —
    holds accumulated entries state, Load More fetches client-side via api.client). This is
    the "client-side accumulator" pattern from research Pattern 7:

      ```typescript
      'use client';

      import { useEffect, useState } from 'react';
      import { Button } from '@/components/ui/button';
      import { fetchEntriesClient } from '@/lib/api.client';
      import { EntriesTable } from './EntriesTable';
      import type { DataEntry, EntryFilters } from '@/types/api';

      export function EntriesClient({
        initialEntries,
        initialNextCursor,
        sourceMap,
        filters,
      }: {
        initialEntries: DataEntry[];
        initialNextCursor: string | null;
        sourceMap: Record<string, string>;
        filters: EntryFilters;
      }): React.JSX.Element {
        const [entries, setEntries] = useState<DataEntry[]>(initialEntries);
        const [nextCursor, setNextCursor] = useState<string | null>(initialNextCursor);
        const [loading, setLoading] = useState(false);
        const [error, setError] = useState<string | null>(null);

        // Reset accumulator when server-side filters change (URL-driven).
        // Filter identity key forces re-init when any filter param changes.
        const filtersKey = JSON.stringify({
          category: filters.category, sourceId: filters.sourceId,
          from: filters.from, to: filters.to,
        });

        useEffect(() => {
          setEntries(initialEntries);
          setNextCursor(initialNextCursor);
          setError(null);
        }, [filtersKey, initialEntries, initialNextCursor]);

        async function loadMore(): Promise<void> {
          if (!nextCursor || loading) return;
          setLoading(true);
          setError(null);
          try {
            const result = await fetchEntriesClient({ ...filters, cursor: nextCursor, limit: 20 });
            setEntries((prev) => [...prev, ...result.items]);
            setNextCursor(result.nextCursor);
          } catch (err) {
            setError(err instanceof Error ? err.message : 'Failed to load more entries');
          } finally {
            setLoading(false);
          }
        }

        return (
          <div className="space-y-4">
            <EntriesTable entries={entries} sourceMap={sourceMap} />
            {error && (
              <p className="text-sm text-red-600 text-center">{error}</p>
            )}
            {nextCursor && (
              <div className="flex justify-center">
                <Button
                  onClick={loadMore}
                  disabled={loading}
                  variant="default"
                  className="w-full md:w-[200px]"
                >
                  {loading ? 'Loading…' : 'Load 20 more'}
                </Button>
              </div>
            )}
          </div>
        );
      }
      ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser searchParams ↔ Next.js server | Untrusted query string values flow into `fetchEntries` |
| Browser ↔ .NET API | EntriesClient fetches from `NEXT_PUBLIC_API_URL` directly via `fetchEntriesClient` |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-10 | Tampering (XSS) | DataEntry.payload contains arbitrary JSON from crawled sources, rendered into the table | mitigate | `JSON.stringify(payload)` produces a safe string. Rendered as text inside `<code>` (not `dangerouslySetInnerHTML`). React auto-escapes. |
| T-07-11 | Tampering | searchParams accept any string from the URL bar; an attacker could craft `category=<script>...` | mitigate | searchParams are passed to `fetchEntries` as URLSearchParams values — automatically URL-encoded. The .NET API treats them as opaque filter strings. No XSS surface in our code (values are not interpolated into HTML). |
| T-07-12 | Information disclosure | `cursor` from URL exposed in browser; could a malicious cursor probe non-public records? | accept | Cursor is opaque base64 keyset token; the API enforces filter clauses (`WHERE category=X`) regardless of cursor. No record-level access control needed (single-user personal project, REQUIREMENTS.md explicitly out-of-scope auth). |
| T-07-13 | DoS | Attacker crafts `limit=999999` in URL | mitigate | The .NET API caps `limit` at 100 server-side (`Math.Clamp(limit, 1, 100)` in EntriesEndpoints.cs line 30). Our client never sets > 20. |
| T-07-14 | Information disclosure | `from`/`to` Date constructor parses arbitrary strings — could throw or produce wrong ISO | mitigate | `new Date(v)` returns Invalid Date for bad input — `.toISOString()` then throws. Wrap in try/catch is overkill; user only enters via `<input type="date">` which restricts format. Acceptable. |
</threat_model>
