    Step 1: Create `apps/dashboard/components/layout/NavLinks.tsx` (`'use client'` required for
    `usePathname`):

      ```typescript
      'use client';

      import Link from 'next/link';
      import { usePathname } from 'next/navigation';
      import { Database, Settings2, ListTodo } from 'lucide-react';
      import { cn } from '@/lib/utils';

      const NAV_ITEMS = [
        { href: '/entries', label: 'Entries', Icon: Database },
        { href: '/sources', label: 'Sources', Icon: Settings2 },
        { href: '/jobs', label: 'Jobs', Icon: ListTodo },
      ] as const;

      export function NavLinks({ onNavigate }: { onNavigate?: () => void }): React.JSX.Element {
        const pathname = usePathname();
        return (
          <nav className="flex flex-col gap-1 p-3">
            {NAV_ITEMS.map(({ href, label, Icon }) => {
              const isActive = pathname === href || pathname.startsWith(`${href}/`);
              return (
                <Link
                  key={href}
                  href={href}
                  onClick={onNavigate}
                  className={cn(
                    'flex items-center gap-3 rounded-md px-3 py-2.5 text-sm min-h-[44px] transition-colors',
                    isActive
                      ? 'bg-zinc-900 text-white font-medium'
                      : 'text-zinc-700 hover:bg-zinc-200',
                  )}
                >
                  <Icon size={16} aria-hidden="true" />
                  <span>{label}</span>
                </Link>
              );
            })}
          </nav>
        );
      }
      ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-06 | Tampering (XSS) | Nav link labels rendered into JSX | mitigate | Labels are hardcoded constants in `NAV_ITEMS`, never user input. React auto-escapes JSX text — no `dangerouslySetInnerHTML`. |
| T-07-07 | Information disclosure | `error.tsx` could leak stack traces with internal paths to the browser | mitigate | UI shows only static copy ("Failed to load data..."). The `error` object is logged via `console.error` in `useEffect` but NOT rendered into the DOM. Next.js production build also strips error.message from client. |
| T-07-08 | Spoofing | A malicious page using the dashboard's iframe could manipulate sidebar nav | accept | No clickjacking protection added (CSP/X-Frame-Options would be Phase 10 deployment concern, not this phase). Personal project, no third-party embedding. |
| T-07-09 | Denial of service | Open redirect via root page | mitigate | `redirect('/entries')` is a hardcoded internal path — no user-controlled input. |
</threat_model>
