```csharp
namespace WebCrawlerApi.Parsers;
using System.Text.Json;

public class LolParser(ILogger<LolParser> logger) : IContentParser
{
    public Task<IReadOnlyList<ParsedEntry>> ParseAsync(
        string rawContent, string sourceId, CancellationToken ct = default)
    {
        var results = new List<ParsedEntry>();
        try
        {
            using var doc = JsonDocument.Parse(rawContent);
            // Navigate to champion data array — adapt path based on LoLWorker.ts
            // Try __NEXT_DATA__ path first, then flat array
            JsonElement champions;
            var root = doc.RootElement;

            if (root.TryGetProperty("props", out var props) &&
                props.TryGetProperty("pageProps", out var pageProps))
            {
                // u.gg __NEXT_DATA__ structure
                if (pageProps.TryGetProperty("data", out champions)) { /* found */ }
                else if (pageProps.TryGetProperty("tierList", out champions)) { /* alt */ }
                else
                {
                    logger.LogWarning("LolParser: no data found in pageProps. SourceId={SourceId}", sourceId);
                    return Task.FromResult<IReadOnlyList<ParsedEntry>>(results);
                }
            }
            else if (root.ValueKind == JsonValueKind.Array)
            {
                champions = root;
            }
            else if (root.TryGetProperty("data", out champions)) { /* direct data */ }
            else
            {
                logger.LogWarning("LolParser: unrecognized JSON structure. SourceId={SourceId}", sourceId);
                return Task.FromResult<IReadOnlyList<ParsedEntry>>(results);
            }

            foreach (var champ in champions.EnumerateArray())
            {
                var name = champ.TryGetProperty("championName", out var cn) ? cn.GetString()
                    : champ.TryGetProperty("champion", out var c) ? c.GetString()
                    : champ.TryGetProperty("name", out var n) ? n.GetString()
                    : null;

                var role = champ.TryGetProperty("role", out var r) ? r.GetString()
                    : champ.TryGetProperty("position", out var p) ? p.GetString()
                    : "unknown";

                if (name == null)
                {
                    logger.LogWarning("LolParser: champion entry missing name, skipping. SourceId={SourceId}", sourceId);
                    continue;
                }

                var entryKey = $"champion_{name.ToLowerInvariant()}_{role?.ToLowerInvariant() ?? "unknown"}";

                var tier = champ.TryGetProperty("tier", out var t) ? t.GetString() : null;
                var winRate = champ.TryGetProperty("winRate", out var wr) && wr.ValueKind == JsonValueKind.Number
                    ? wr.GetDouble() : (double?)null;
                var pickRate = champ.TryGetProperty("pickRate", out var pr) && pr.ValueKind == JsonValueKind.Number
                    ? pr.GetDouble() : (double?)null;
                var patch = champ.TryGetProperty("patch", out var pa) ? pa.GetString() : null;

                var payload = new
                {
                    champion = name,
                    role,
                    tier,
                    win_rate = winRate,
                    pick_rate = pickRate,
                    patch
                };

                results.Add(new ParsedEntry(sourceId, entryKey, "game", payload));
            }

            logger.LogInformation("LolParser: produced {Count} entries. SourceId={SourceId}", results.Count, sourceId);
        }
        catch (Exception ex)
        {
            logger.LogWarning(ex, "LolParser: parse failed. SourceId={SourceId}", sourceId);
        }

        return Task.FromResult<IReadOnlyList<ParsedEntry>>(results);
    }
}
```

```csharp
namespace WebCrawlerApi.Parsers;
using System.Text.Json;

public class AniListParser(ILogger<AniListParser> logger) : IContentParser
{
    public Task<IReadOnlyList<ParsedEntry>> ParseAsync(
        string rawContent, string sourceId, CancellationToken ct = default)
    {
        var results = new List<ParsedEntry>();
        try
        {
            using var doc = JsonDocument.Parse(rawContent);
            var root = doc.RootElement;

            // Navigate AniList GraphQL response structure
            JsonElement mediaList;
            if (root.TryGetProperty("data", out var data))
            {
                if (data.TryGetProperty("Page", out var page) &&
                    page.TryGetProperty("media", out mediaList)) { /* standard */ }
                else if (data.TryGetProperty("media", out mediaList)) { /* alt */ }
                else
                {
                    logger.LogWarning("AniListParser: no media list in response. SourceId={SourceId}", sourceId);
                    return Task.FromResult<IReadOnlyList<ParsedEntry>>(results);
                }
            }
            else if (root.ValueKind == JsonValueKind.Array)
            {
                mediaList = root;
            }
            else
            {
                logger.LogWarning("AniListParser: unrecognized structure. SourceId={SourceId}", sourceId);
                return Task.FromResult<IReadOnlyList<ParsedEntry>>(results);
            }

            foreach (var anime in mediaList.EnumerateArray())
            {
                if (!anime.TryGetProperty("id", out var idEl))
                {
                    logger.LogWarning("AniListParser: anime missing 'id', skipping. SourceId={SourceId}", sourceId);
                    continue;
                }

                var animeId = idEl.GetInt32();
                var entryKey = $"anime_{animeId}";

                // Title: prefer English, fallback to romaji
                string? title = null;
                if (anime.TryGetProperty("title", out var titleObj))
                {
                    title = titleObj.TryGetProperty("english", out var eng) && eng.ValueKind == JsonValueKind.String
                        ? eng.GetString()
                        : titleObj.TryGetProperty("romaji", out var rom) ? rom.GetString() : null;
                }

                if (title == null)
                {
                    logger.LogWarning("AniListParser: anime {Id} missing title, skipping. SourceId={SourceId}", animeId, sourceId);
                    continue;
                }

                int? episode = null;
                string? airDate = null;
                if (anime.TryGetProperty("nextAiringEpisode", out var nextEp) &&
                    nextEp.ValueKind != JsonValueKind.Null)
                {
                    episode = nextEp.TryGetProperty("episode", out var ep) ? ep.GetInt32() : null;
                    if (nextEp.TryGetProperty("airingAt", out var airingAt))
                    {
                        airDate = DateTimeOffset.FromUnixTimeSeconds(airingAt.GetInt64())
                            .ToString("yyyy-MM-dd");
                    }
                }

                var status = anime.TryGetProperty("status", out var st)
                    ? st.GetString()?.ToLowerInvariant().Replace("_", " ") : null;
                var score = anime.TryGetProperty("averageScore", out var sc) && sc.ValueKind == JsonValueKind.Number
                    ? sc.GetDouble() / 10.0 : (double?)null;  // AniList uses 0-100, SCHEMA shows 0-10

                var payload = new
                {
                    title,
                    episode,
                    air_date = airDate,
                    status,
                    mal_score = score
                };

                results.Add(new ParsedEntry(sourceId, entryKey, "anime", payload));
            }

            logger.LogInformation("AniListParser: produced {Count} entries. SourceId={SourceId}", results.Count, sourceId);
        }
        catch (Exception ex)
        {
            logger.LogWarning(ex, "AniListParser: parse failed. SourceId={SourceId}", sourceId);
        }

        return Task.FromResult<IReadOnlyList<ParsedEntry>>(results);
    }
}
```

```csharp
namespace WebCrawlerApi.Parsers;
using System.Text.Json;

public class MangaDexParser(ILogger<MangaDexParser> logger) : IContentParser
{
    public Task<IReadOnlyList<ParsedEntry>> ParseAsync(
        string rawContent, string sourceId, CancellationToken ct = default)
    {
        var results = new List<ParsedEntry>();
        try
        {
            using var doc = JsonDocument.Parse(rawContent);
            var root = doc.RootElement;

            if (!root.TryGetProperty("data", out var dataArr) || dataArr.ValueKind != JsonValueKind.Array)
            {
                logger.LogWarning("MangaDexParser: no 'data' array in response. SourceId={SourceId}", sourceId);
                return Task.FromResult<IReadOnlyList<ParsedEntry>>(results);
            }

            foreach (var chapter in dataArr.EnumerateArray())
            {
                if (!chapter.TryGetProperty("id", out var idEl))
                {
                    logger.LogWarning("MangaDexParser: chapter missing 'id', skipping. SourceId={SourceId}", sourceId);
                    continue;
                }

                var chapterId = idEl.GetString();
                var entryKey = $"chapter_{chapterId}";

                string? chapterNum = null, title = null, volume = null, publishAt = null, language = null;
                if (chapter.TryGetProperty("attributes", out var attrs))
                {
                    chapterNum = attrs.TryGetProperty("chapter", out var cn) ? cn.GetString() : null;
                    title = attrs.TryGetProperty("title", out var t) ? t.GetString() : null;
                    volume = attrs.TryGetProperty("volume", out var v) ? v.GetString() : null;
                    publishAt = attrs.TryGetProperty("publishAt", out var pa) ? pa.GetString() : null;
                    language = attrs.TryGetProperty("translatedLanguage", out var lang) ? lang.GetString() : null;
                }

                // Extract manga title from relationships
                string? mangaTitle = null;
                if (chapter.TryGetProperty("relationships", out var rels))
                {
                    foreach (var rel in rels.EnumerateArray())
                    {
                        if (rel.TryGetProperty("type", out var relType) && relType.GetString() == "manga" &&
                            rel.TryGetProperty("attributes", out var relAttrs) &&
                            relAttrs.TryGetProperty("title", out var relTitle))
                        {
                            mangaTitle = relTitle.TryGetProperty("en", out var en) ? en.GetString()
                                : relTitle.TryGetProperty("ja", out var ja) ? ja.GetString()
                                : null;
                            break;
                        }
                    }
                }

                var payload = new
                {
                    manga_title = mangaTitle,
                    chapter = chapterNum,
                    title,
                    volume,
                    language,
                    publish_date = publishAt
                };

                results.Add(new ParsedEntry(sourceId, entryKey, "manga", payload));
            }

            logger.LogInformation("MangaDexParser: produced {Count} entries. SourceId={SourceId}", results.Count, sourceId);
        }
        catch (Exception ex)
        {
            logger.LogWarning(ex, "MangaDexParser: parse failed. SourceId={SourceId}", sourceId);
        }

        return Task.FromResult<IReadOnlyList<ParsedEntry>>(results);
    }
}
```
